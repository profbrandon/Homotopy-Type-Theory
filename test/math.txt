
define 
  Bool  : @,
  true  : Bool,
  false : Bool,

  not false := true,
  not true  := false,

  or true  b := true,
  or false b := b,

  and true  b := b,
  and false b := false,

  xor false b := b,
  xor true  b := not b,

  Nat  : @,
  n0   : Nat,
  succ : Nat -> Nat,

  pred n0       := n0,
  pred (succ n) := n,

  add n0 m := m,
  add (succ n) m := succ m
in
